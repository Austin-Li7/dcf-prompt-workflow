**Goal:** Build a Python orchestration system that runs the v5 7-step AI valuation pipeline end-to-end for any ticker, starting with MSFT Step 1–4 validation.

**Architecture:** Step classes (BaseStep subclasses) are orchestrated by PipelineRunner. Each step calls three LLM providers (Claude for generation, Gemini Flash for checkpoints, GPT-4o for CSF audit), evaluates a gate, and passes a StepOutput forward. SEC filings are pre-fetched once per ticker and cached locally.

**Tech Stack:** Python 3.11+, anthropic, google-generativeai, openai, httpx, pyyaml, rich, pytest, pytest-mock

```python
# config.py
from __future__ import annotations
from dataclasses import dataclass, field
from pathlib import Path
import yaml


@dataclass
class AppConfig:
    api_keys: dict[str, str]
    models: dict[str, str]
    csf_review_mode: dict[str, str]
    tickers: list[str] = field(default_factory=list)


def load_config(path: Path | str = "config.yaml") -> AppConfig:
    with open(path) as f:
        data = yaml.safe_load(f)
    return AppConfig(
        api_keys=data["api_keys"],
        models=data["models"],
        csf_review_mode=data["csf_review_mode"],
        tickers=data.get("tickers", []),
    )
```

```python
from anthropic import Anthropic
from providers.base import LLMProvider


class AnthropicProvider(LLMProvider):
    def __init__(self, api_key: str, model: str, max_tokens: int = 8192):
        self._client = Anthropic(api_key=api_key)
        self._model = model
        self._max_tokens = max_tokens

    def call(self, prompt: str) -> str:
        msg = self._client.messages.create(
            model=self._model,
            max_tokens=self._max_tokens,
            messages=[{"role": "user", "content": prompt}],
        )
        return msg.content[0].text
```

```python
# utils/gate_evaluator.py
"""
One parser per checkpoint per step. Each returns GateVerdict.
Gate evaluation functions combine verdicts into a GateResult.
"""
from __future__ import annotations
import re
from datetime import datetime, timezone
from steps.models import GateVerdict, GateResult


# ─── Step 1 ─────────────────────────────────────────────────────────────────

def parse_step1_ckpt_a(response: str, expected_count: int) -> GateVerdict:
    """Response should be a single integer matching the segment count."""
    stripped = response.strip()
    match = re.search(r"\b(\d+)\b", stripped)
    if not match:
        return GateVerdict(False, f"Non-integer response: '{stripped}'")
    count = int(match.group(1))
    if count == expected_count:
        return GateVerdict(True, f"Segment count {count} matches expected {expected_count}")
    return GateVerdict(False, f"Count mismatch: got {count}, expected {expected_count}")


def parse_step1_ckpt_b(response: str) -> GateVerdict:
    """Response is numbered lines of FOUND / NOT_FOUND."""
    lines = [l.strip() for l in response.strip().splitlines() if l.strip()]
    total = len(lines)
    not_found = [l for l in lines if "NOT_FOUND" in l]
    found = total - len(not_found)
    if not_found:
        return GateVerdict(False, f"{found}/{total} FOUND; NOT_FOUND: {not_found}")
    return GateVerdict(True, f"{total}/{total} FOUND")


def parse_step1_ckpt_c(response: str) -> GateVerdict:
    """Response is numbered lines of FOUND_VERBATIM / FOUND_PARAPHRASED / NOT_FOUND."""
    lines = [l.strip() for l in response.strip().splitlines() if l.strip()]
    not_found = [l for l in lines if "NOT_FOUND" in l]
    if not_found:
        return GateVerdict(False, f"NOT_FOUND snippets: {not_found}")
    verbatim = sum(1 for l in lines if "FOUND_VERBATIM" in l)
    paraphrased = sum(1 for l in lines if "FOUND_PARAPHRASED" in l)
    return GateVerdict(True, f"{verbatim} VERBATIM, {paraphrased} PARAPHRASED of {len(lines)}")


def evaluate_step1_gate(
    ckpt_a: GateVerdict,
    ckpt_b: GateVerdict,
    ckpt_c: GateVerdict,
    csf_reviewed: bool,
) -> GateResult:
    verdicts = {"A": ckpt_a.detail, "B": ckpt_b.detail, "C": ckpt_c.detail}
    csf_status = "reviewed" if csf_reviewed else "pending"
    ts = datetime.now(timezone.utc).isoformat()

    if not ckpt_a.passed:
        return GateResult("FAIL", verdicts, csf_status, f"Checkpoint A failed: {ckpt_a.detail}", ts)
    if not ckpt_b.passed:
        return GateResult("FAIL", verdicts, csf_status, f"Checkpoint B failed: {ckpt_b.detail}", ts)
    if not ckpt_c.passed:
        return GateResult("CONDITIONAL_PASS", verdicts, csf_status, f"Checkpoint C: {ckpt_c.detail}", ts)
    return GateResult("PASS", verdicts, csf_status, None, ts)


# ─── Step 2 ─────────────────────────────────────────────────────────────────

def parse_step2_ckpt_a(response: str) -> GateVerdict:
    """Response is MATCH or MISMATCH."""
    text = response.strip().upper()
    if "MATCH" in text and "MISMATCH" not in text:
        return GateVerdict(True, "Rollup MATCH")
    return GateVerdict(False, f"Rollup MISMATCH: {response.strip()}")


def parse_step2_ckpt_b(response: str) -> GateVerdict:
    """3 lines of MATCH / MISMATCH spot-checks."""
```

```python
# utils/claim_tracker.py
from __future__ import annotations
from dataclasses import dataclass, field


EVIDENCE_ORDER: dict[str, int] = {
    "UNSUPPORTED": 0,
    "WEAK_INFERENCE": 1,
    "STRONG_INFERENCE": 2,
    "DISCLOSED": 3,
}


@dataclass
class ClaimRecord:
    claim_id: str
    step_id: str
    evidence_level: str
    history: list[dict] = field(default_factory=list)


class ClaimTracker:
    def __init__(self, ticker: str):
        self.ticker = ticker
        self.registry: dict[str, ClaimRecord] = {}

    def register(self, claim_id: str, step_id: str, evidence_level: str) -> None:
        if claim_id not in self.registry:
            self.registry[claim_id] = ClaimRecord(claim_id, step_id, evidence_level)
        else:
            record = self.registry[claim_id]
            record.history.append({
                "step_id": record.step_id,
                "evidence_level": record.evidence_level,
            })
            record.step_id = step_id
            record.evidence_level = evidence_level

    def get_laundering_candidates(self) -> list[dict]:
        candidates = []
        for claim_id, record in self.registry.items():
            if not record.history:
                continue
            original_level = record.history[0]["evidence_level"]
            current_level = record.evidence_level
            if EVIDENCE_ORDER.get(current_level, 0) > EVIDENCE_ORDER.get(original_level, 0):
                candidates.append({
                    "claim_id": claim_id,
                    "original_level": original_level,
                    "original_step": record.history[0]["step_id"],
                    "upgraded_to": current_level,
                    "upgraded_in_step": record.step_id,
                })
        return candidates

    def summary_report(self) -> str:
        candidates = self.get_laundering_candidates()
        if not candidates:
            return f"[{self.ticker}] Claim tracker: no laundering candidates found."
        lines = [f"[{self.ticker}] LAUNDERING CANDIDATES ({len(candidates)}):"]
        for c in candidates:
            lines.append(
                f"  {c['claim_id']}: {c['original_level']} (step {c['original_step']}) "
                f"→ {c['upgraded_to']} (step {c['upgraded_in_step']})"
            )
        return "\n".join(lines)
```

```python
# utils/filing_fetcher.py
from __future__ import annotations
import json
import re
from pathlib import Path
import httpx


class FilingNotFoundError(Exception):
    pass


class FilingFetcher:
    TICKERS_URL = "https://www.sec.gov/files/company_tickers.json"
    SUBMISSIONS_URL = "https://data.sec.gov/submissions/CIK{cik:010d}.json"
    FILING_BASE = "https://www.sec.gov/Archives/edgar/data/{cik}/{accession_clean}/{doc}"
    HEADERS = {"User-Agent": "DCF Pipeline research@checkit.com"}

    def __init__(self, cache_dir: Path | str):
        self.cache_dir = Path(cache_dir)

    # ── CIK lookup ────────────────────────────────────────────────────────

    def get_cik(self, ticker: str) -> int:
        resp = httpx.get(self.TICKERS_URL, headers=self.HEADERS, timeout=30)
        resp.raise_for_status()
        data = json.loads(resp.text)
        ticker_upper = ticker.upper()
        for entry in data.values():
            if entry["ticker"].upper() == ticker_upper:
                return int(entry["cik_str"])
        raise FilingNotFoundError(f"CIK not found for ticker: {ticker}")

    # ── Filing index ──────────────────────────────────────────────────────

    def get_latest_filing_info(self, ticker: str, form_type: str) -> tuple[str, str]:
        """Returns (accession_number, primary_document) for the latest filing of form_type."""
        cik = self.get_cik(ticker)
        url = self.SUBMISSIONS_URL.format(cik=cik)
        resp = httpx.get(url, headers=self.HEADERS, timeout=30)
        resp.raise_for_status()
        subs = json.loads(resp.text)
        recent = subs["filings"]["recent"]
        for i, form in enumerate(recent["form"]):
            if form == form_type:
                return recent["accessionNumber"][i], recent["primaryDocument"][i]
        raise FilingNotFoundError(f"No {form_type} found for {ticker}")

    # ── Download & cache ──────────────────────────────────────────────────

    def fetch_and_cache(self, ticker: str, form_type: str, cache_key: str) -> str:
        """Download filing, strip HTML tags, cache to disk, return text."""
        ticker_dir = self.cache_dir / ticker
        ticker_dir.mkdir(parents=True, exist_ok=True)
        cache_file = ticker_dir / f"{cache_key}.txt"

        if cache_file.exists():
            return cache_file.read_text(encoding="utf-8")

        cik = self.get_cik(ticker)
        accession, doc = self.get_latest_filing_info(ticker, form_type)
        accession_clean = accession.replace("-", "")
        url = self.FILING_BASE.format(cik=cik, accession_clean=accession_clean, doc=doc)
        resp = httpx.get(url, headers=self.HEADERS, timeout=60, follow_redirects=True)
        resp.raise_for_status()

        text = self._strip_html(resp.text)
        cache_file.write_text(text, encoding="utf-8")
        return text

    def _strip_html(self, html: str) -> str:
        text = re.sub(r"<[^>]+>", " ", html)
        text = re.sub(r"&nbsp;", " ", text)
        text = re.sub(r"&amp;", "&", text)
        text = re.sub(r"&lt;", "<", text)
        text = re.sub(r"&gt;", ">", text)
        text = re.sub(r" {2,}", " ", text)
        text = re.sub(r"\n{3,}", "\n\n", text)
        return text.strip()

    # ── Section extraction ────────────────────────────────────────────────

    def extract_section(self, filing_text: str, section_name: str, max_chars: int = 8000) -> str:
        """Extract text between section_name header and the next all-caps header."""
        pattern = re.compile(
            rf"(?i){re.escape(section_name)}\s*\n(.*?)(?=\n[A-Z][A-Z\s]{{10,}}\n|\Z)",
            re.DOTALL,
        )
        match = pattern.search(filing_text)
        if not match:
            return f"[SECTION NOT FOUND: {section_name}]"
        return match.group(1)[:max_chars].strip()

    def get_section(self, ticker: str, cache_key: str, section_name: str) -> str:
        """Convenience: load cached filing and extract a section."""
        cache_file = self.cache_dir / ticker / f"{cache_key}.txt"
        if not cache_file.exists():
            raise FilingNotFoundError(f"Cached filing not found: {cache_file}")
        text = cache_file.read_text(encoding="utf-8")
        return self.extract_section(text, section_name)

    # ── Cache check ────────────────────────────────────────────────────────

    def is_cached(self, ticker: str, cache_key: str) -> bool:
        return (self.cache_dir / ticker / f"{cache_key}.txt").exists()

    # ── Prefetch all filings for a ticker ─────────────────────────────────

    def prefetch(self, ticker: str) -> dict[str, str]:
        """Download 10-K, 10-Q, and latest 8-K for ticker. Returns {cache_key: path}."""
        result = {}
        for form, key in [("10-K", "10k_latest"), ("10-Q", "10q_latest")]:
            try:
                self.fetch_and_cache(ticker, form, key)
                result[key] = str(self.cache_dir / ticker / f"{key}.txt")
            except Exception as e:
                result[key] = f"ERROR: {e}"
        return result
```

```python
One-time script: parse V5_DCF/v5_RUNNABLE_PROMPT_SET.md
into individual .txt prompt templates.

Run from dcf_pipeline/:
    python prompts/parse_prompts.py
"""
import re
from pathlib import Path

SOURCE = Path("../V5_DCF/v5_RUNNABLE_PROMPT_SET.md")
OUTPUT_DIR = Path("prompts")

SECTION_HEADER_RE = re.compile(r"# ──────────────────────────\s*\n(.*?)\n# ──────────────────────────", re.DOTALL)
PROMPT_CONTENT_RE = re.compile(r'"""\s*\n(.*?)"""', re.DOTALL)

# Map section title keywords → output filename
FILENAME_MAP = {
    "STEP 1 GENERATION": "step1_generation",
    "STEP 1 CHECKPOINT A": "step1_checkpoint_a",
    "STEP 1 CHECKPOINT B": "step1_checkpoint_b",
    "STEP 1 CHECKPOINT C": "step1_checkpoint_c",
    "STEP 1 CLAIM-SUPPORT-FIT": "step1_csf_audit",
    "STEP 2 GENERATION": "step2_generation",
    "STEP 2 CHECKPOINT A": "step2_checkpoint_a",
    "STEP 2 CHECKPOINT B": "step2_checkpoint_b",
    "STEP 2 CHECKPOINT C": "step2_checkpoint_c",
    "STEP 2 CLAIM-SUPPORT-FIT": "step2_csf_audit",
    "STEP 3 GENERATION": "step3_generation",
    "STEP 3 CHECKPOINT A": "step3_checkpoint_a",
    "STEP 3 CHECKPOINT B": "step3_checkpoint_b",
    "STEP 3 CLAIM-SUPPORT-FIT": "step3_csf_audit",
    "STEP 4 GENERATION": "step4_generation",
    "STEP 4 CHECKPOINT A": "step4_checkpoint_a",
    "STEP 4 CHECKPOINT B": "step4_checkpoint_b",
    "STEP 4 CLAIM-SUPPORT-FIT": "step4_csf_audit",
    "STEP 4.5 GENERATION": "step4_5_generation",
    "STEP 4.5 CHECKPOINT": "step4_5_checkpoint_a",
    "STEP 5 GENERATION": "step5_generation",
    "STEP 5 CHECKPOINT A": "step5_checkpoint_a",
    "STEP 5 CHECKPOINT B": "step5_checkpoint_b",
    "STEP 5 CHECKPOINT C": "step5_checkpoint_c",
    "STEP 5 CHECKPOINT D": "step5_checkpoint_d",
    "STEP 5 CLAIM-SUPPORT-FIT": "step5_csf_audit",
    "STEP 7 CHECKPOINT A": "step7_checkpoint_a",
    "STEP 7 CHECKPOINT B": "step7_checkpoint_b",
    "PIPELINE AUDIT": "pipeline_audit",
}


def get_filename(header_text: str) -> str | None:
    header_upper = header_text.upper()
    for keyword, filename in FILENAME_MAP.items():
        if keyword in header_upper:
            return filename
    return None


def parse_and_write():
    source_text = SOURCE.read_text(encoding="utf-8")

    # Split into chunks between section dividers
    chunks = re.split(r"\n# ──────────────────────────\n", source_text)

    written = []
    i = 0
    while i < len(chunks):
        chunk = chunks[i]
        # Look for a header comment block (lines starting with #)
        header_lines = [l for l in chunk.splitlines() if l.startswith("#") and "──" not in l]
        header_text = " ".join(header_lines)
        filename = get_filename(header_text)

        if filename:
            # Find the triple-quoted prompt in this or next chunk
            search_text = chunk
            if i + 1 < len(chunks):
                search_text += chunks[i + 1]
            match = PROMPT_CONTENT_RE.search(search_text)
            if match:
                content = match.group(1).strip()
                out_path = OUTPUT_DIR / f"{filename}.txt"
                out_path.write_text(content, encoding="utf-8")
                written.append(filename)
                print(f"  ✓ {filename}.txt")
        i += 1

    print(f"\nWrote {len(written)} prompt files to {OUTPUT_DIR}/")


if __name__ == "__main__":
    parse_and_write()
```

```python
# steps/base_step.py
from __future__ import annotations
import json
import sys
from abc import ABC, abstractmethod
from dataclasses import asdict
from pathlib import Path

from providers.base import LLMProvider
from steps.models import StepContext, StepOutput, GateResult


class BaseStep(ABC):
    step_id: str  # override in subclass: "1", "2", "4_5", etc.

    def __init__(
        self,
        generation_provider: LLMProvider,
        checkpoint_provider: LLMProvider,
        csf_provider: LLMProvider,
        csf_review_mode: str = "auto",  # "auto" | "interactive"
    ):
        self._gen = generation_provider
        self._ckpt = checkpoint_provider
        self._csf = csf_provider
        self._csf_mode = csf_review_mode

    # ── Public API ────────────────────────────────────────────────────────

    def run(self, context: StepContext) -> StepOutput:
        out_dir = Path(context.output_dir)

        # 1. Generate
        gen_prompt = self._generation_prompt(context)
        gen_output = self._gen.call(gen_prompt)
        self._write(out_dir / f"step{self.step_id}_generation.md", gen_output)

        # 2. Checkpoints
        checkpoint_specs = self._checkpoint_prompts(context, gen_output)
        checkpoint_results: dict[str, str] = {}
        for spec in checkpoint_specs:
            key = spec["key"]
            prompt = spec["prompt"]
            result = self._ckpt.call(prompt)
            checkpoint_results[key] = result
            self._write(out_dir / f"step{self.step_id}_checkpoint_{key.lower()}.md", result)

        # 3. CSF Audit
        csf_prompt = self._csf_prompt(context, gen_output)
        csf_output = self._csf.call(csf_prompt)
        self._write(out_dir / f"step{self.step_id}_csf_audit.md", csf_output)

        # 4. Handle CSF review mode
        csf_reviewed = self._handle_csf_review(context, csf_output)

        # 5. Evaluate gate
        gate = self._parse_gate(checkpoint_results, csf_output)
        if not csf_reviewed:
            gate.csf_status = "pending"

        # 6. Save gate JSON
        gate_dict = asdict(gate)
        gate_dict["step"] = self.step_id
        gate_dict["ticker"] = context.ticker
        self._write(
            out_dir / f"step{self.step_id}_gate.json",
            json.dumps(gate_dict, indent=2),
        )

        # 7. Extract structured claims
        claims_json = self._extract_claims_json(gen_output)

        return StepOutput(
            step_id=self.step_id,
            ticker=context.ticker,
            raw_text=gen_output,
            claims_json=claims_json,
            gate=gate,
            checkpoint_outputs=checkpoint_results,
            csf_output=csf_output,
        )

    # ── Abstract methods (override in each step) ──────────────────────────

    @abstractmethod
    def _generation_prompt(self, context: StepContext) -> str: ...

    @abstractmethod
    def _checkpoint_prompts(self, context: StepContext, gen_output: str) -> list[dict]:
        """Return list of {key: str, prompt: str} dicts."""
        ...

    @abstractmethod
    def _csf_prompt(self, context: StepContext, gen_output: str) -> str: ...

    @abstractmethod
    def _parse_gate(self, checkpoint_results: dict[str, str], csf_result: str) -> GateResult: ...

    @abstractmethod
    def _extract_claims_json(self, gen_output: str) -> dict: ...

    # ── Helpers ───────────────────────────────────────────────────────────

    def _handle_csf_review(self, context: StepContext, csf_output: str) -> bool:
        if self._csf_mode != "interactive":
            return True
        out_dir = Path(context.output_dir)
        audit_path = out_dir / f"step{self.step_id}_csf_audit.md"
        print(f"\n{'='*60}")
        print(f"[Step {self.step_id} CSF Audit — interactive review]")
        print(f"Review: {audit_path}")
        print(f"{'='*60}")
        print(csf_output[:2000])
        print(f"\nPress Enter to PASS, or type FAIL to stop pipeline: ", end="")
        response = input().strip().upper()
        if response == "FAIL":
            return False
        return True

    @staticmethod
    def _write(path: Path, content: str) -> None:
        path.parent.mkdir(parents=True, exist_ok=True)
        path.write_text(content, encoding="utf-8")

    @staticmethod
    def _load_prompt(prompt_file: str) -> str:
        """Load a prompt template from prompts/ directory."""
        path = Path(__file__).parent.parent / "prompts" / prompt_file
        if not path.exists():
            raise FileNotFoundError(f"Prompt template not found: {path}")
        return path.read_text(encoding="utf-8")
```

import json

```python
class Step1(BaseStep):
    step_id = "1"

    def _generation_prompt(self, context: StepContext) -> str:
        template = self._load_prompt("step1_generation.txt")
        return template.replace("[INSERT TICKER]", context.ticker)

    def _checkpoint_prompts(self, context: StepContext, gen_output: str) -> list[dict]:
        filing_section = self._get_segment_section(context)
        segment_names = self._extract_segment_names(gen_output)
        claim_snippets = self._extract_random_claims(gen_output, n=5)

        ckpt_a_template = self._load_prompt("step1_checkpoint_a.txt")
        ckpt_a = ckpt_a_template.replace(
            "[PASTE: 10-K segment information footnote]", filing_section
        )

        ckpt_b_template = self._load_prompt("step1_checkpoint_b.txt")
        segments_list = "\n".join(f"{i+1}. {s}" for i, s in enumerate(segment_names))
        ckpt_b = (
            ckpt_b_template
            .replace(
                "1. [segment_1]\n2. [segment_2]\n3. [segment_3]",
                segments_list,
            )
            .replace(
                "[PASTE: 10-K segment disclosure paragraph]",
                filing_section,
            )
        )

        ckpt_c_template = self._load_prompt("step1_checkpoint_c.txt")
        ckpt_c = ckpt_c_template.replace(
            "[repeat for 5 randomly selected claims]",
            claim_snippets,
        ).replace(
            "[PASTE: relevant 10-K pages]",
            filing_section,
        )

        return [
            {"key": "A", "prompt": ckpt_a},
            {"key": "B", "prompt": ckpt_b},
            {"key": "C", "prompt": ckpt_c},
        ]

    def _csf_prompt(self, context: StepContext, gen_output: str) -> str:
        template = self._load_prompt("step1_csf_audit.txt")
        all_claims = self._extract_all_claims_text(gen_output)
        return template.replace("[PASTE: all claims from Step 1 Phase 1 output]", all_claims)

    def _parse_gate(self, checkpoint_results: dict[str, str], csf_result: str) -> GateResult:
        # Expected segment count from generation output's self_check
        expected_count = self._last_expected_count  # set during _checkpoint_prompts
        ckpt_a = parse_step1_ckpt_a(checkpoint_results.get("A", ""), expected_count)
        ckpt_b = parse_step1_ckpt_b(checkpoint_results.get("B", ""))
        ckpt_c = parse_step1_ckpt_c(checkpoint_results.get("C", ""))
        return evaluate_step1_gate(ckpt_a, ckpt_b, ckpt_c, csf_reviewed=True)

    def _extract_claims_json(self, gen_output: str) -> dict:
        match = re.search(r"```json\s*(\{.*?\})\s*```", gen_output, re.DOTALL)
        if not match:
            return {}
        try:
            return json.loads(match.group(1))
        except json.JSONDecodeError:
            return {}

    # ── Private helpers ──────────────────────────────────────────────────

    def _get_segment_section(self, context: StepContext) -> str:
        fetcher = FilingFetcher(cache_dir=Path(context.filing_dir).parent)
        for section_name in ["SEGMENT INFORMATION", "Note 19", "Operating Segments"]:
            text = fetcher.get_section(context.ticker, "10k_latest", section_name)
            if "[SECTION NOT FOUND" not in text:
                return text
        return "[Segment section not found in cached 10-K]"

    def _extract_segment_names(self, gen_output: str) -> list[str]:
        """Extract segment names from generation output's architecture JSON."""
        data = self._extract_claims_json(gen_output)
        arch = data.get("architecture", [])
        if arch:
            return [item.get("segment", "") for item in arch if item.get("segment")]
        # Fallback: regex on TEXT claims with segment pattern
        matches = re.findall(r'"segment":\s*"([^"]+)"', gen_output)
        return matches[:10]

    def _extract_random_claims(self, gen_output: str, n: int = 5) -> str:
        """Extract up to n CLAIM blocks from Phase 1 output for Checkpoint C."""
        claims = re.findall(
            r"(CLAIM:.*?EVIDENCE_LEVEL:.*?)(?=\nCLAIM:|\nPhase 2|\Z)",

            re.DOTALL,
        )
        selected = claims[:n]
        return "\n\n".join(c.strip() for c in selected)

    def _extract_all_claims_text(self, gen_output: str) -> str:
        """Extract all CLAIM blocks for the CSF audit."""
        claims = re.findall(
            r"(CLAIM:.*?EVIDENCE_LEVEL:.*?)(?=\nCLAIM:|\nPhase 2|\Z)",
            gen_output,
            re.DOTALL,
        )
        return "\n\n".join(c.strip() for c in claims)

    def _checkpoint_prompts(self, context: StepContext, gen_output: str) -> list[dict]:
        # Extract expected segment count from self_check in generation output
        data = self._extract_claims_json(gen_output)
        self._last_expected_count = data.get("self_check", {}).get("segments_in_10K_footnote", 3)

        filing_section = self._get_segment_section(context)
        segment_names = self._extract_segment_names(gen_output)
        claim_snippets = self._extract_random_claims(gen_output, n=5)

        ckpt_a_template = self._load_prompt("step1_checkpoint_a.txt")
        ckpt_a = ckpt_a_template.replace(
            "[PASTE: 10-K segment information footnote]", filing_section
```

```python
    def _build_providers(self):
        keys = self._cfg.api_keys
        models = self._cfg.models
        self._gen_provider = AnthropicProvider(keys["anthropic"], models["generation"])
        self._ckpt_provider = GoogleProvider(keys["google"], models["checkpoint"])
        self._csf_provider = OpenAIProvider(keys["openai"], models["csf_audit"])

    def _build_steps(self):
        mode = self._cfg.csf_review_mode
        self._step1 = Step1(
            generation_provider=self._gen_provider,
            checkpoint_provider=self._ckpt_provider,
            csf_provider=self._csf_provider,
            csf_review_mode=mode.get("step1", "auto"),
        )
        # Steps 2-4 added as implemented

    def _make_context(self, ticker: str) -> StepContext:
        out_dir = self._output_base / ticker
        filing_dir = out_dir / "filings"
        out_dir.mkdir(parents=True, exist_ok=True)
        filing_dir.mkdir(parents=True, exist_ok=True)
        return StepContext(
            ticker=ticker,
            output_dir=str(out_dir),
            filing_dir=str(filing_dir),
        )

    def run_step(self, step_id: str, context: StepContext) -> StepOutput:
        step_map = {"1": self._step1}
        if step_id not in step_map:
            raise ValueError(f"Step {step_id} not yet implemented")
        step = step_map[step_id]
        return step.run(context)

    def run_ticker(self, ticker: str, steps: list[str] | None = None) -> dict:
        steps_to_run = steps or AVAILABLE_STEPS
        context = self._make_context(ticker)
        results: dict[str, StepOutput] = {}

        console.print(Panel(f"[bold]Running pipeline for {ticker}[/bold]", expand=False))

        for step_id in steps_to_run:
            console.print(f"\n[cyan]→ Step {step_id}[/cyan]")
            try:
                output = self.run_step(step_id, context)
            except ValueError as e:
                console.print(f"  [yellow]Skipping: {e}[/yellow]")
                continue

            results[step_id] = output

            # Attach to context for next step
            if step_id == "1":
                context.step1 = output
            elif step_id == "2":
                context.step2 = output
            elif step_id == "3":
                context.step3 = output
            elif step_id == "4":
                context.step4 = output

            if output.gate.decision == "FAIL":
                console.print(f"  [red]✗ GATE FAIL: {output.gate.blocking_reason}[/red]")
                console.print(f"  [red]Pipeline stopped. Human review required.[/red]")
                return {
                    "ticker": ticker,
                    "stopped_at": step_id,
                    "reason": output.gate.blocking_reason,
                    "completed_steps": list(results.keys()),
                }

            console.print(f"  [green]✓ Gate: {output.gate.decision}[/green]")

        console.print(f"\n[bold green]Pipeline complete for {ticker}[/bold green]")
        return {
            "ticker": ticker,
            "stopped_at": None,
            "reason": None,
            "completed_steps": list(results.keys()),
        }

    def prefetch_filings(self, ticker: str) -> None:
        console.print(f"[cyan]Prefetching filings for {ticker}...[/cyan]")
        fetcher = FilingFetcher(cache_dir=self._output_base / ticker / "filings")
        result = fetcher.prefetch(ticker)
        for key, path in result.items():
            if path.startswith("ERROR"):
                console.print(f"  [red]✗ {key}: {path}[/red]")
            else:
                console.print(f"  [green]✓ {key}[/green]")

    parser = argparse.ArgumentParser(description="DCF valuation pipeline")
    parser.add_argument("--ticker", required=False, help="Company ticker (e.g. MSFT)")
    parser.add_argument("--steps", nargs="*", help="Steps to run (default: all available)")
    parser.add_argument("--prefetch", metavar="TICKER", help="Prefetch SEC filings only")
```
